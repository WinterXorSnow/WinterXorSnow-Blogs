# [C - Distance Indicators](https://atcoder.jp/contests/abc417/tasks/abc417_c)

## 解题思路
考虑将原来的式子变为$A_i + i= j - A_j$，然后就可以倒着做，对于每一个$i$，只需要去看后面有多少个位置满足这个式子就可以了

## AC代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define pii pair<int,int>
#define fi first
#define se second
namespace WinterXorSnow{
	const int N = 2e5 + 10;
	map<int,int> mp;
	int a[N];
	int n;
	void work(){
		cin >>  n;
		for(int i=1;i<=n;i++) cin >> a[i];
		ll ans = 0;
		for(int i=n;i>=1;i--){
			ans += 1ll * mp[i + a[i]];
			mp[i - a[i]] ++ ;
		}
		cout << ans;
	}
}
int main()
{
	int T;
	T = 1;
	// cin >> T;
	while(T -- ) WinterXorSnow::work();
	return 0;
}
```
# [D - Takahashi's Expectation](https://atcoder.jp/contests/abc417/tasks/abc417_d)
~~感觉这道题做起来比E题难是怎么回事……~~
## 解题思路
$10000$的$N$和$5e5$的$Q$意味着这道题如果想要彻底的在线做是不现实的

首先想到的是记忆化，但是很显然，范围相当大的$x$也没有给我们记忆化的机会

而后可以注意到，$a,b,p$全都很小，最大只有$500$

因此，我们不难意识到，当$x$被减到$500$左右时，就一定是在上下波动。然后，再仔细想一想，就会发现波动的范围也是很小的，在$0$~$1000$之间

>[!info] 证明
>倘若$x$大于$500$，那么很显然，一定是在一直减少$b$
>
>对于$x$小于等于$500$的情况，考虑$x$刚好为500，且$a$也刚好为$500$，那么相加，只会到$1000$
>
>由此，我们就可以得出$x$当进入到"波动"的情况后，其范围就是$0$到$1000$

有了上面的结论，我们就可以先预处理出来对于每一个$i(1 \le i \le n)$，当前的$x$为$j$的最终答案(很显然，这个最终答案也是在$0$~$1000$之中的)，这个过程直接记忆化+模拟就行了(具体的，可以倒着做，然后递推解决)

当然，很容易想到，$x$来到$1000$的时刻是不一样的，而在其$\le 1000$之前，必然是要一路减去$b$的，因此，可以二分在哪一个时刻，$x$的值小于$1000$
## AC代码
注：这道题要实现起来还是要考虑一些细节的，因此最好不要直接看代码
``` cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define pii pair<int,int>
#define fi first
#define se second
namespace WinterXorSnow{
	const int N = 10005,M = 1005;
	int n;
	int p[N],a[N],b[N];
	int sum[N];
	int f[N][M];
	void work(){
		cin >> n;
		for(int i=1;i<=n;i++){
			cin >> p[i] >> a[i] >> b[i];
			sum[i] = sum[i-1] + b[i];
		}
		for(int i=0;i<=1000;i++){
			f[n+1][i] = i;
		}
		for(int i=n;i>=1;i--){
			for(int j=0;j<=1000;j++){
				if(j > p[i])
					f[i][j] = f[i+1][max(0,j-b[i])];
				else{
					f[i][j] = f[i+1][j+a[i]];
				}
			}
		}
		int Q;
		cin >> Q;
		while(Q -- ){
			int x;
			cin >> x;
			int fnd = -1;
			int l=1,r=n;
			while(l <= r){
				int mid = (l + r) >> 1;
				if(x - sum[mid-1] <= 1000){
					fnd = mid;
					r = mid - 1;
				}
				else l = mid + 1;
			}
			if(x <= 1000) cout << f[1][x] << "\n";
			else if(fnd == -1) cout << x - sum[n] << "\n";
			else{
				cout << f[fnd][x - sum[fnd-1]] << "\n";
			}
		}
	}
}
int main()
{
	IOS;
	int T;
	T = 1;
	// cin >> T;
	while(T -- ) WinterXorSnow::work();
	return 0;
}
```

# [E - A Path in A Dictionary](https://atcoder.jp/contests/abc417/tasks/abc417_e)
## 解题思路
因为本题要求的是字典序最小，因此考虑贪心，只用一个劲的去走值最小的点就行了

注意：不能回溯

> [!info]  正确性证明
> **首先考虑不回溯(退标记)的正确性**
> 
> 如果一个节点退了标记，但是后面还被用到，那么此前在处理这个节点的时候就是可行的，这与需要回退(即不可行)矛盾，因此不回溯是正确的
> 
> **然后考虑为什么不用考虑路径长度**
> 
> 因为不可能有两条路径存在完全的包含关系。每条路径都必须要以$y$结尾，并且，路径中不存在相同的点，这意味着对于长度不同的路径，其最少有一位不同（就是在较短路径的末尾）

## AC代码
``` cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define pii pair<int,int>
#define fi first
#define se second
namespace WinterXorSnow{
	const int N = 1005;
	vector<int> G[N];
	bool vis[N];
	int pre[N];
	int n,m,x,y;
	void dfs(int u){
		for(int v : G[u]){
			if(vis[v]) continue;
			vis[v] = 1;
			pre[v] = u;
			dfs(v);
		}
	}
	void print(int x){
		if(x == -1) return;
		print(pre[x]);
		cout << x << ' ';
		return ;
	}
	void work(){
		memset(pre,0,sizeof pre);
		memset(vis,0,sizeof vis);
		cin >> n >> m >> x >> y;
		for(int i=1;i<=n;i++) G[i].clear();
		for(int i=1;i<=m;i++){
			int u,v;
			cin >> u >> v;
			G[u].push_back(v);
			G[v].push_back(u);
		}
		for(int i=1;i<=n;i++) sort(G[i].begin(),G[i].end());
		pre[x] = -1;
		vis[x] = 1;
		dfs(x);
		// for(int i=1;i<=n;i++) cout << pre[i] << ' ';
		// cout << "\n";
		print(y);
		cout << "\n";
	}
}
int main()
{
	// IOS;
	int T;
	T = 1;
	cin >> T;
	while(T -- ) WinterXorSnow::work();
	return 0;
}
```

# [F - Random Gathering](https://atcoder.jp/contests/abc417/tasks/abc417_f)
~~赛时看到期望直接被吓退了（悲）~~
## 解题思路
看到这种区间修改的题，很容易想到使用**线段树**进行维护

然后考虑期望的计算：

这道题期望的计算其实是比较显然的，将期望表示为$E$就是$E_i= \sum_{j=l}^r{E_j} \times \frac{1}{r-l+1}$，不难理解

因此，这道题几乎就是裸的**线段树**
## AC代码

``` cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define pii pair<int,int>
#define fi first
#define se second
namespace WinterXorSnow{
	const ll mod = 998244353;
	const int N = 2e5 + 10;
	int n,m;
	ll tre[N*4],lazy[N*4];
	void exgcd(ll a,ll b,ll &x,ll &y){
		if(b == 0){x=1,y=0;return;}
		exgcd(b,a%b,y,x);y -= x * (a/b);
	}
	ll inv(ll a,ll b){
		ll x,y;
		exgcd(a,b,x,y);
		return (x%b+b)%b;
	}
	void pushup(int p){
		tre[p] = tre[p*2] + tre[p*2+1]; tre[p]%=mod;return ;
	}
	void pushdown(int pl,int pr,int p){
		if(lazy[p] != -1){
			int mid = (pl + pr) >> 1;
			lazy[p*2] = lazy[p];
			lazy[p*2+1] = lazy[p];
			tre[p*2] = 1ll * lazy[p] * (mid - pl + 1) % mod;
			// tre[p*2] %= mod;
			tre[p*2+1] = 1ll * lazy[p] * (pr - mid) % mod;
			// tre[p*2+1] %= mod;
			lazy[p] = -1;
		}
	}
	int a[N];
	void build(int pl,int pr,int p){
		lazy[p] = -1;
		if(pl == pr){
			tre[p] = a[pl];
			return ;
		}
		int mid = (pl + pr) >> 1;
		build(pl,mid,p*2);
		build(mid+1,pr,p*2+1);
		pushup(p);return ;
	}
	void update(int L,int R,int pl,int pr,int p,ll k){
		if(L <= pl && pr <= R){
			tre[p] = 1ll * (pr - pl + 1) * k % mod;
			tre[p] %= mod;
			lazy[p] = k;
			return ;
		}
		int mid = (pl + pr) >> 1;
		pushdown(pl,pr,p);
		if(L <= mid) update(L,R,pl,mid,p*2,k);
		if(R > mid) update(L,R,mid+1,pr,p*2+1,k);
		pushup(p);
	}
	ll query(int L,int R,int pl,int pr,int p){
		if(L <= pl && pr <= R) return tre[p];
		pushdown(pl,pr,p);
		int mid = (pl + pr) >> 1;
		ll re = 0;
		if(L <= mid) re += query(L,R,pl,mid,p*2);
		if(R > mid) re += query(L,R,mid+1,pr,p*2+1);
		re %= mod;
		pushup(p);
		return re;
	}
	void work(){
		cin >> n >> m;
		for(int i=1;i<=n;i++) cin >> a[i];
		build(1,n,1);
		for(int i=1;i<=m;i++){
			int l,r;
			cin >> l >> r;
			ll sum = query(l,r,1,n,1) % mod;
			// cout << sum << "\n";
			ll tmp = sum * inv(r-l+1,mod) % mod;
			tmp %= mod;
			// cout << inv(r-l+1,mod) << "\n";
			// cout << tmp << "\n";
			update(l,r,1,n,1,tmp);
		}
		for(int i=1;i<=n;i++){
			cout << query(i,i,1,n,1) % mod << " ";
		}
		return ;
	}
}
int main()
{
	IOS;
	int T;
	T = 1;
	// cin >> T;
	while(T -- ) WinterXorSnow::work();
	return 0;
}
```
