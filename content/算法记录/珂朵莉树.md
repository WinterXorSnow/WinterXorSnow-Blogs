### 节点的的定义
实际上，珂朵莉树的本质就是把连续且相同的数字并到一起。（这也就是为什么其需要数据里有次数较多的推平操作）。

因此节点的定义如下：
```cpp
struct node{
	int l,r;
	mutable LL v;
	node(int L,int R=-1,LL v=0) : l(L),r(R),v(v) {};
	bool operator < (const node& o) const{ return l < o.l;}
};
```

注意一下语法知识点：`mutable` 的意思是易变的，这个能够无视 `const` 的限制，使得我们能直接修改 `set` 内节点的值。

还有一个要注意的点是：在`node(int L,int R=-1,LL v=0) : l(L),r(R),v(v) {};`这句话中，`v` 是必须要赋初值的。因为 c++ 要求我们若要赋初值，则必须要是一段后缀。也就是一个赋了初值，后面都要赋初值。

### 基本操作——分裂(Split)
这个操作是在把序列 `[l,r]` 分成 `[l,pos-1]` 与 `[pos,r]`。其中，满足 $pos \in [l,r]$。
```cpp
IT split(int pos){
	IT it = st.lower_bound(node(pos));
	if(it != st.end() && it->l == pos) return it;
	it -- ;
	int L,R;
	LL val;
	L = it->l;R = it->r;val = it->v;
	st.erase(it);
	st.insert(node(L,pos-1,val));
	return st.insert(node(pos,R,val)).fi;
}
```

具体解释下：

`IT it = st.lower_bound(node(pos));` 

这个是在找第一个 $l \ge pos$ 的区间。

`if(it != st.end() && it->l == pos) return it;`

已经不可再分，直接返回。

`return st.insert(node(pos,R,val)).fi;`

这里又有语法知识点：`set` 的 `insert` 操作是有返回值的，为 `pair` 类型。其中，第一维为迭代器，第二维为 `bool` 类型，用来反映是否插入成功。

### 关键操作——区间推平(assign)
珂朵莉树的核心。实际上很简单。
```cpp
void assign(int l,int r,LL val){
	IT itr = split(r+1),itl = split(l);
	st.erase(itl,itr);
	st.insert(node(l,r,val));
}
```

注意点有点多，讲下：
- `IT itr = split(r+1),itl = split(l);` 此处只能这样写，如果交换顺序，会导致 `itl` 可能被 `itr` 影响，指向滚木，导致 *RE* 。
- `itr = split(r+1)` ，这里不要写成 `r`。

然后说下语法：`set` 的 `erase` 操作是能传入迭代器的，具体的，`erase(itl,itr)` 表示删除了 `set` 中 `[itl,itr)` 的节点。~~（ `std` 库的人好喜欢左闭右开）~~

### Code
```cpp
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define LL long long
#define fi first
#define se second
#define IT set<node>::iterator
LL qpow(LL a,LL b,LL mod){
	if(b == 0) return 1;
	if(b == 1) return a;
	LL re = qpow(a,b/2,mod);
	re*= re;
	re %= mod;
	if(b & 1ll) re *= a;
	return re % mod;
}
struct ODT{
	struct node{
		int l,r;
		mutable LL v;
		node(int L,int R=-1,LL v=0) : l(L),r(R),v(v) {};
		bool operator < (const node& o) const{ return l < o.l;}
	};
	set<node> st;
	IT split(int pos){
		IT it = st.lower_bound(node(pos));
		if(it != st.end() && it->l == pos) return it;
		it -- ;
		int L,R;
		LL val;
		L = it->l;
		R = it->r;
		val = it->v;
		st.erase(it);
		st.insert(node(L,pos-1,val));
		return st.insert(node(pos,R,val)).fi;
	}
	void assign(int l,int r,LL val){
		IT itr = split(r+1),itl = split(l);
		st.erase(itl,itr);
		st.insert(node(l,r,val));
	}
	void add(int l,int r,LL x){
		IT itr = split(r+1),itl = split(l);
		for(;itl!=itr;itl++)
			itl->v += x;
		return ;
	}
	LL rank(int l,int r,int k){
		IT itr = split(r+1),itl = split(l);
		vector< pair<LL,int> > vec;
		for(;itl!=itr;itl++)
			vec.push_back({itl->v,itl->r - itl->l + 1});
		sort(vec.begin(),vec.end());
		for(int i=0;i<vec.size();i++){
			k -= vec[i].se;
			if(k <= 0) return vec[i].fi;
		}
		return -1;
	}
	LL sum(int l,int r,LL x,LL mod){
		IT itr = split(r+1),itl = split(l);
		LL ans = 0;
		for(;itl!=itr;itl++){
			ans += 1ll * (itl->r - itl->l + 1) * qpow(itl->v % mod,x,mod) % mod;
			ans %= mod;
		}
		return ans;
	}
}tre;
int n,m;
LL seed,vmax;
LL rnd(){
	LL ret = seed;
	seed = (seed * 7 + 13) % 1000000007;
	return ret;
}
int main()
{
	IOS;
	cin >> n >> m >> seed >> vmax;
	for (int i=1;i<=n;i++){
		int x;
		x = (rnd() % vmax) + 1;
		tre.st.insert(ODT::node(i,i,x));
	}
	tre.st.insert(ODT::node(n+1,n+1,0));//保护节点，防RE
	for(int i=1;i<=m;i++){
		int op,l,r;
		op = (rnd() % 4) + 1;
	    l = (rnd() % n) + 1;
	    r = (rnd() % n) + 1;

	    if (l > r){
	        swap(l, r);
	    }
	    LL x,y;
	    if (op == 3){
	       x = (rnd() % (r - l + 1)) + 1;
	    }
	    else{
	       x = (rnd() % vmax) + 1;
	    }

	    if (op == 4){
	        y = (rnd() % vmax) + 1;
	    }
	    if(op == 1) tre.add(l,r,x);
	    if(op == 2) tre.assign(l,r,x);
	    if(op == 3) cout << tre.rank(l,r,x) << "\n";
	    if(op == 4) cout << tre.sum(l,r,x,y) << "\n";
	}
	return 0;
}
```
