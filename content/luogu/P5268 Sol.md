#莫队 #容斥 #差分 #题解

[题目链接](https://www.luogu.com.cn/problem/P5268)

这种题都想不出来，确有问题。

转化一下题目中给的式子，变成：

$$
\begin{aligned} 
\sum_0^N get[(1,r_1,x)-(1,l_1-1,x)] \times get[(1,r_2,x)-(1,l_2-1,x)]
\end{aligned}
$$

然后拆成四个求和，做完了。

## Code
``` cpp
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define LL long long
#define fi first
#define se second
const int N = 5e4 + 10;
struct ques{
	int x,y;
	int id;
	int fh;
}q[N << 2];
int n,Q;
int a[N];
int tot;
void add(int x,int y,int id,int fh){
	if(x == 0 || y ==0 ) return ;
	tot ++ ;
	q[tot] = {x,y,id,fh};
	return ;
}
bool cmp(ques x,ques y){
	if(x.x == y.x) return x.y < y.y;
	return x.x < y.x;
}
bool cmp2(ques x,ques y){
	if(x.y ==  y.y) return x.x < y.x;
	return x.y < y.y;
};
LL ans[N];
namespace Sort{
int L[N << 2],R[N << 2],p[N << 2];
int max_T = 0;
void init(){
	int T = sqrt(n);
	memset(L,0x3f,sizeof L);
	for(int i=1;i<=tot;i++){
		p[i] = q[i].x / T + 1;
		max_T = max(max_T,p[i]);
		L[p[i]] = min(L[p[i]],i);
		R[p[i]] = max(R[p[i]],i);
	}
	for(int i=1;i<=max_T;i++){
		if(L[i] > R[i]) continue;
		sort(q+L[i]+1,q+R[i]+1,cmp2);
	}
}
}
LL cnt1[N],cnt2[N];
namespace solve{
	int l,r;
	LL cnt = 0;
	void add(int x,int type){
		if(type == 1){
			cnt += cnt2[a[x]];
			cnt1[a[x]] ++ ;
		}
		else{
			cnt += cnt1[a[x]];
			cnt2[a[x]] ++ ;
		}
		return; 
	}
	void del(int x,int type){
		if(type == 1){
			cnt -= cnt2[a[x]];
			cnt1[a[x]] -- ;
		}
		else{
			cnt -= cnt1[a[x]];
			cnt2[a[x]] -- ;
		}
		return; 
	}
	void modui(){
		l = r = 0;
		for(int i=1;i<=tot;i++){
			while(l < q[i].x) add(++l,1);
			while(l > q[i].x) del(l--,1);
			while(r < q[i].y) add(++r,2);
			while(r > q[i].y) del(r--,2);
			ans[q[i].id] += q[i].fh * cnt;
		}
		return ;
	}
}
int main()
{
	IOS;
	cin >> n;
	for(int i=1;i<=n;i++)
		cin >> a[i];
	cin >> Q;
	for(int i=1;i<=Q;i++){
		int x,y,xx,yy;
		cin >> x >> y >> xx >> yy;
		add(y,yy,i,1);
		add(x-1,xx-1,i,1);
		add(x-1,yy,i,-1);
		add(y,xx-1,i,-1);
	}
	sort(q+1,q+1+tot,cmp);
	Sort::init();
	solve::modui();
	for(int i=1;i<=Q;i++)
		cout << ans[i] << "\n";
	return 0;
}
```
