#分块 #题解 

[题目链接](https://www.luogu.com.cn/problem/P14400)

神秘分块。

区间问题，首先想到线段树/分块这类的东西。但是这道题线段树不好想，因此考虑分块。

观察一下，发现每一次查询就是在把区间 $[l,r]$ 以内的最大值取出。考虑分块维护。

- 对于整块：考虑使用大根堆，然后乱搞就好了。
- 对于分块：发现更新非常难搞，考虑使用小根堆，用来记录每一次加进整块的堆里面的值。然后遍历这个块内的点就好了。

时间复杂度 $O \sqrt n \log n$。

## Code
```cpp
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define LL long long
#define fi first
#define se second
const int N = 4e5 + 10,B = 633;
int n,Q;
priority_queue<int> qmax[B];
priority_queue< int, vector<int>, greater<int> > tag[B];
priority_queue< int , vector<int> ,greater<int> > gunmu;
int cnt;
int a[N];
int L[B],R[B],bl[N];
priority_queue< int > make_max(int l,int r){
	priority_queue< int > re;
	for(int i=l;i<=r;i++)
		re.push(a[i]);
	return re;
}
void rebuild(int x){
	if(tag[x].empty()) return ;
	for(int i=L[x];i<=R[x];i++){
		if(a[i] > tag[x].top()){
			tag[x].push(a[i]);
			a[i] = tag[x].top();
			tag[x].pop();
		}
	}
	tag[x] = gunmu;
	return ;
}
int query(int l,int r,int x){
	int lx = bl[l],rx = bl[r];
	if(lx == rx){
		rebuild(lx);
		for(int i=l;i<=r;i++){
			if(a[i] > x) swap(a[i],x);
		}
		qmax[lx] = make_max(L[lx],R[lx]);
		return x;
	}
	rebuild(lx);
	rebuild(rx);
	for(int i=l;i<=R[lx];i++){
		if(a[i] > x) swap(a[i],x);
	}
	qmax[lx] = make_max(L[lx],R[lx]);
	for(int i=lx+1;i<=rx-1;i++){
		if(qmax[i].top() > x){
			int t = qmax[i].top();
			qmax[i].pop();
			qmax[i].push(x);
			tag[i].push(x);
			x = t;
		}
	}
	for(int i=L[rx];i<=r;i++){
		if(a[i] > x) swap(a[i],x);
	}
	qmax[rx] = make_max(L[rx],R[rx]);
	return x;
}
int main()
{
	IOS;
	cin >> n >> Q;
	cnt = (n + B - 1) / B;
	for(int i=1;i<=cnt;i++){
		L[i] = R[i-1] + 1;
		R[i] = B * i;
	}
	R[cnt] = n;
	for(int i=1;i<=n;i++){
		cin >> a[i];
		bl[i] = (i + B - 1) / B;
	}
	for(int i=1;i<=cnt;i++){
		qmax[i] = make_max(L[i],R[i]);
	}
	while(Q -- ){
		int S,T,P;
		cin >> S >> T >> P;
		if(S <= T){
			cout << query(S,T,P);
		}
		else{
			cout << query(1,T,query(S,n,P));
		}
		cout << "\n";
	}
	return 0;
}
```
