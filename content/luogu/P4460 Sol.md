#DP  #题解 

[题目链接](https://www.luogu.com.cn/problem/P4460)

简单**状压DP**，严肃独立写出。

## 思路
看到 $n$ 很小，考虑状压DP。

考虑状态间的直接转移，发现如果只是常规的定义 $f[i]$ 根本无法转移，于是在定义一个 $j$ ，表示末尾在 $j$，接下来预处理出两点之间的点，就很好转移了。

时间复杂度$O(n^22^n)$。

## Code
注意一下横坐标相等的情况就好了
```cpp
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define LL long long
#define fi first
#define se second
const int N = 20,M = (1<<19)|5;
const double eps = 1e-6;
const LL mod = 1e8 + 7;
int n;
pair<int,int> a[N];
int dis[N][N];
inline double calc(pair<int,int> x,pair<int,int> y){
	return 1.0 * (1.0 * (y.se - x.se)) / (1.0 * (y.fi - x.fi));
}
inline bool chk(double x,double y){
	if(fabs(x-y) <= eps) return 1;
	else return 0;
}
inline double lenth(pair<int,int> x,pair<int,int> y){
	return sqrt(1.0 * (x.fi - y.fi) * (x.fi - y.fi) + 1.0 * (x.se - y.se) * (x.se - y.se));
}
int mycount(int x){ return __builtin_popcount(x);}
int f[M][N];
int main()
{
	IOS;
	cin >> n;
	for(int i=1;i<=n;i++)
		cin >> a[i].fi >> a[i].se;
	sort(a+1,a+1+n);
	for(int i=1;i<=n;i++){
		for(int j=i+1;j<=n;j++){
			for(int k=i+1;k<=n;k++){
				if(i == k) continue;
				if(j == k) continue;
				if(a[i].fi == a[j].fi){
					if(j > k) dis[i][j] += (1 << k-1);
				}
				else if(chk(calc(a[i],a[j]),calc(a[i],a[k])) && lenth(a[i],a[j]) > lenth(a[i],a[k])){
					dis[i][j] += (1 << k-1);
				}
			}
			dis[j][i] = dis[i][j];
		}
	}
	for(int i=1;i<=n;i++)
		f[1<<i-1][i-1] = 1;
	for(int i=1;i<(1<<n);i++){
		for(int j=0;j<n;j++){
			if(f[i][j] == 0) continue;
			for(int k=0;k<n;k++){
				if((1 << k) & i) continue;
				int w = dis[j+1][k+1];
				if((w & i) == w){
					f[i+(1<<k)][k] += f[i][j];
					f[i+(1<<k)][k] %= mod;
				}
			}
		}
	}
	LL ans = 0;
	for(int i=1;i<(1<<n);i++){
		if(mycount(i) >= 4){
			for(int j=0;j<n;j++){
				ans += 1ll * f[i][j];
				ans %= mod;
			}
		}
	}
	cout << ans;
	return 0;
}
```
