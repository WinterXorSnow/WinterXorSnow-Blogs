#题解 
# P5500 [LnOI2019] 真正的 OIer 从不女装 Sol
[题目链接](https://www.luogu.com.cn/problem/P5500)

发现性质+线段树，兑

## 解题思路

考虑一次女装的贡献：

$a_1,a_2,a_3,\dots,a_n$ 在选择 $p$ 进行反转后，变为：

$a_p,a_{p-1},a_{p-2},\dots,a_2,a_1,a_n,a_{n-1},a_{n-2},\dots,a_{p+1}$。

显然反转答案不变，因此变为：

$a_{p+1},a_{p+2},\dots,a_{n-1},a_n,a_1,a_2,a_3,\dots,a_{p-1},a_{p}$，观察一下，这不就是把区间变成一个环，每次操作，相当于转一下环。

因此，我们可以得出结论：**女装一次等价于女装无数次**。所以结论早就写在题目里了。

## Code
简单线段树。兑。有些细节，仔细一点。
```cPP
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false);cin.tie(0),cout.tie(0)
#define File(s) freopen(s".in","r",stdin);freopen(s".out","w",stdout)
#define LL long long
#define fi first
#define se second
const int N = 2e5 + 10;
int n,Q;
int a[N];
struct segtree{
	struct node{
		int lcol,rcol;
		int col;//col of the all range
		int lmax,rmax;// left max,right max
		int ans;
		node(){
			lcol = rcol = -1;
			col = -1;
			ans = 0;
			lmax = 0;
			rmax = 0;
		}
		node operator + (const node oth){
			node x;
			if(col != oth.col || col == -1 || oth.col == -1) x.col = -1;
			else x.col = col;
			x.lcol = lcol;
			x.rcol = oth.rcol;
			if(col == oth.lcol && col != -1) x.lmax = lmax + oth.lmax;
			else x.lmax = lmax;
			if(rcol == oth.col && oth.col != -1) x.rmax = rmax + oth.rmax;
			else x.rmax = oth.rmax;
			x.ans = max(max(x.lmax,x.rmax),max(ans,oth.ans));
			if(rcol == oth.lcol && rcol != -1 && oth.lcol != -1) x.ans = max(x.ans,rmax + oth.lmax);
			return x;
		}
	}tre[N << 2];
	int lzy[N << 2];
	void pushup(int p){
		tre[p] = tre[p*2] + tre[p*2+1];
		return ;
	}
	void pushdown(int pl,int pr,int p){
		if(lzy[p] ==0 ) return ;
		int mid = pl + pr >> 1;
		lzy[p*2] = lzy[p*2+1] = lzy[p];
		tre[p*2].lcol = tre[p*2].rcol = tre[p*2].col = lzy[p];
		tre[p*2].lmax = tre[p*2].rmax = tre[p*2].ans = mid-pl+1;
		tre[p*2+1].lcol = tre[p*2+1].rcol = tre[p*2+1].col = lzy[p];
		tre[p*2+1].lmax = tre[p*2+1].rmax = tre[p*2+1].ans = pr-mid;
		lzy[p] = 0;
		return ;
	}
	void build(int pl,int pr,int p){
		if(pl == pr){
			tre[p].lcol = tre[p].rcol = tre[p].col = a[pl];
			tre[p].lmax = tre[p].rmax = tre[p].ans = 1;
			return ;
		}
		int mid = pl + pr >> 1;
		build(pl,mid,p*2);
		build(mid+1,pr,p*2+1);
		tre[p] = tre[p*2] + tre[p*2+1];
		return ;
	}
	void update(int L,int R,int pl,int pr,int p,int x){
		if(L <= pl && pr <= R){
			lzy[p] = x;
			tre[p].col = tre[p].lcol = tre[p].rcol = x;
			tre[p].lmax = tre[p].rmax = tre[p].ans = pr-pl+1;
			return ;
		}
		pushdown(pl,pr,p);
		int mid = pl + pr >> 1;
		if(L <= mid) update(L,R,pl,mid,p*2,x);
		if(R > mid) update(L,R,mid+1,pr,p*2+1,x);
		pushup(p);
		return ;
	}
	node query(int L,int R,int pl,int pr,int p){
		if(L <= pl && pr <= R) return tre[p];
		pushdown(pl,pr,p);
		int mid = pl + pr >> 1;
		if(L <= mid && mid < R) return query(L,R,pl,mid,p*2) + query(L,R,mid+1,pr,p*2+1);
		else if(L <= mid) return query(L,R,pl,mid,p*2);
		else return query(L,R,mid+1,pr,p*2+1);
	}
	node tmp;
}seg;
int main()
{
	IOS;
	cin >> n >> Q;
	for(int i=1;i<=n;i++)
		cin >> a[i];
	seg.build(1,n,1);
	while(Q -- ){
		char c;
		cin >> c;
		if(c == 'R'){
			int l,r,x;
			cin >> l >> r >> x;
			seg.update(l,r,1,n,1,x);
		}
		else{
			int l,r,k;
			cin >> l >> r >> k;
			if(k == 0) cout << seg.query(l,r,1,n,1).ans << "\n";
			else{
				seg.tmp = seg.query(l,r,1,n,1);
				int ans = seg.tmp.ans;
				if(seg.tmp.lcol == seg.tmp.rcol && seg.tmp.lcol != -1 && seg.tmp.col == -1){
					ans = max(ans,seg.tmp.lmax + seg.tmp.rmax);
				}
				cout << ans << "\n";
			}
		}
	}
	return 0;
}
```
